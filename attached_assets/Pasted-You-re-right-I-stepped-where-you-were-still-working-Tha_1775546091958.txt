  try {
    const id = parseInt(req.params.subAccountId || req.body.subAccountId, 10);
    if (!id) return next();
    const row = await db('protected_accounts').where({ sub_account_id: id }).first();

        traceId: req.traceId || null,

        subAccountId: id,
        message: 'Blocked write to protected account',
        created_at: new Date()

      return res.status(403).json({ error: 'sub_account_protected' });
    }
    next();
