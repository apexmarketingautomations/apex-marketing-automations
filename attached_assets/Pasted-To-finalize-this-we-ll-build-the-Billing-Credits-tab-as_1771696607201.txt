### **2. The "Ghost Markup" Middleware**

This is the "Path" that handles your passive income. It runs in the background of every action.

```javascript
// Middleware to handle usage-based profit
async function processUsageCharge(userId, baseCost) {
    const markupMultiplier = 3.0; // Your 3x markup
    const totalCharge = baseCost * markupMultiplier;
    
    const user = await db.get(`user_${userId}`);
    
    if (user.credits >= totalCharge) {
        // Deduct from user
        user.credits -= totalCharge;
        await db.set(`user_${userId}`, user);
        
        // Log your profit
        const platformProfit = totalCharge - baseCost;
        await db.add('global_platform_profit', platformProfit);
        
        return { success: true, remaining: user.credits };
    } else {
        return { success: false, reason: 'insufficient_funds' };
    }
}

```
