To make this fully hands-off, we need the **Stripe Webhook**. This is the secure "Path" that allows Stripe’s servers to talk to your Replit server. When a payment is successful, Stripe sends a signal, and your code instantly updates the user's credits.

### **The Stripe Webhook Logic**

Add this to your `server.js`. This route is designed to be public but secure, using a signature check to ensure only Stripe can trigger it.

```javascript
const stripe = require('stripe')(process.env.STRIPE_SECRET_KEY);

// Use express.raw() for this specific route to verify the signature
app.post('/api/billing/webhook', express.raw({type: 'application/json'}), async (req, res) => {
    const sig = req.headers['stripe-signature'];
    let event;

    try {
        event = stripe.webhooks.constructEvent(req.body, sig, process.env.STRIPE_WEBHOOK_SECRET);
    } catch (err) {
        console.error(`❌ Webhook Error: ${err.message}`);
        return res.status(400).send(`Webhook Error: ${err.message}`);
    }

    // Handle the checkout.session.completed event
    if (event.type === 'checkout.session.completed') {
        const session = event.data.object;
        const userEmail = session.metadata.user_email; // We pass this when creating the session
        const amountPaid = session.amount_total / 100; // Stripe provides amount in cents

        // ADD CREDITS TO USER
        const user = await db.get(`user_${userEmail}`);
        if (user) {
            user.credits = (user.credits || 0) + amountPaid;
            await db.set(`user_${userEmail}`, user);
            
            console.log(`💰 CREDITS ADDED: $${amountPaid} to ${userEmail}`);
        }
    }

    res.json({received: true});
});

```
