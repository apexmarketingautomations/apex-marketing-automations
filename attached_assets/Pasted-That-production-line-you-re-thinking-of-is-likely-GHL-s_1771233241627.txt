```javascript
// lib/sentinel-rules-engine.ts

export async function processSentinelRules(event) {
  const { type, subAccountId, payload } = event;

  // 1. Check for Negative Sentiment / Low Ratings
  if (type === 'REVIEW_RECEIVED' && payload.rating <= 2) {
    const account = await prisma.subAccount.findUnique({ where: { id: subAccountId } });
    
    // ACTION: Kill the AI outreach to prevent further damage
    await vapi.pauseOutbound(account.vapiId);
    
    // ACTION: Notify the God Mode User
    await sendUrgentAlert(account.agencyOwnerId, `CRITICAL: ${account.name} rating dropped. AI PAUSED.`);
  }
}

```
