### 2. The Stripe Pricing Logic

You’ll run this script once to set up your products in Stripe.

```javascript
// scripts/setup-stripe-tiers.js
const stripe = require('stripe')(process.env.STRIPE_SECRET_KEY);

async function createTiers() {
  const tiers = [
    {
      name: 'Starter AI',
      price: 9700,
      description: 'Perfect for single business owners.'
    },
    {
      name: 'Agency Pro',
      price: 29700,
      description: 'Unlimited sub-accounts & AI Voice Agents.'
    },
    {
      name: 'God Mode (Founder)',
      price: 49700,
      description: 'White-labeling, Ghost SDR, and Custom Branding.'
    }
  ];

  for (const tier of tiers) {
    const product = await stripe.products.create({ name: tier.name, description: tier.description });
    await stripe.prices.create({
      unit_amount: tier.price,
      currency: 'usd',
      recurring: { interval: 'month' },
      product: product.id,
    });
  }
}

```

### 3. The "God Mode" Launch Script

This handles the **60-day deal** logic and the **automatic provisioning** of their first AI Agent so they see value within 60 seconds of paying.

```javascript
// api/checkout/success.js
import { prisma } from '../lib/db';
import { provisionAIAgent } from '../lib/vapi-service';

export default async function handleSuccess(req, res) {
  const { sessionId, userId } = req.body;
  const session = await stripe.checkout.sessions.retrieve(sessionId);

  if (session.payment_status === 'paid') {
    // 1. Apply the 60-Day Deal (Update next billing date in Stripe)
    await stripe.subscriptions.update(session.subscription, {
      trial_end: Math.floor(Date.now() / 1000) + (60 * 24 * 60 * 60), // 60 days from now
    });

    // 2. Provision the Master Sub-Account
    const user = await prisma.user.update({
      where: { id: userId },
      data: { 
        isSubscribed: true, 
        tier: session.metadata.tierName,
        aiCredits: 50.00 // Give them $50 launch credit
      }
    });

    // 3. THE MAGIC: Auto-Provision the first Voice Agent

// lib/snapshot-engine.ts
export async function installSnapshot(targetSubAccountId, snapshotType) {
  const blueprints = {
    'med-spa': {
      workflow: 'Lead Follow-up + Booking',
      botTrainingUrl: 'https://standard-medspa-sop.com',
      vibe: 'clean-glass'
    },
    'real-estate': {
      workflow: 'Property Inquiry + SMS Agent',
      botTrainingUrl: 'https://standard-realestate-faq.com',
      vibe: 'gilded-rose'
    }
  };

  const selected = blueprints[snapshotType];

  // Apply all settings at once to the new sub-account
  await prisma.subAccount.update({
    where: { id: targetSubAccountId },
    data: {
      vibeTheme: selected.vibe,
      activeWorkflows: { create: { name: selected.workflow } }
    }
  });
}

```
