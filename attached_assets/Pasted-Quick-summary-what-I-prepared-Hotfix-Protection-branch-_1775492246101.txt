// server/middleware/ensureNotProtectedAccount.js
const db = require('../db'); // your knex/pg pool
const { parseIntParam } = require('../routes/helpers');

function isMutating(req) {
  const mutatingMethods = ['POST','PUT','PATCH','DELETE'];
  if (mutatingMethods.includes(req.method)) return true;
  // Optional: detect GETs with side-effects by route name header or param
  return false;
}

module.exports = async function ensureNotProtectedAccount(req, res, next) {
  try {
    // find candidate id in params or body
    const candidate = req.params.subAccountId || req.body.subAccountId || req.query.subAccountId;
    if (!candidate) return next(); // not sub-account scoped
    const subAccountId = parseInt(candidate, 10);
    if (Number.isNaN(subAccountId)) return res.status(400).json({ error: 'bad_subAccountId' });

    const row = await db('sub_accounts').where({ id: subAccountId }).first('id','is_protected');
    if (!row) return res.status(404).json({ error: 'sub_account_not_found' });

    if (row.is_protected && isMutating(req)) {
      const traceId = req.traceId || null;
      const userId = req.user?.id || null;
      const ticketId = `P0-${Date.now()}-${Math.floor(Math.random() * 9000) + 1000}`;

      await db('system_logs').insert({
        level: 'security',
        traceId,
        userId,
        subAccountId,
        action: `${req.method} ${req.originalUrl}`,
        meta: JSON.stringify({ msg: 'Blocked write to protected account', ticketId }),
        created_at: new Date()
      });

      // also create a security ticket record if you have a tickets table; else the ticketId is enough
      return res.status(403).json({ error_code: 'sub_account_protected', ticketId });
    }

    // On reads, log an info entry (optional)
    if (row.is_protected && !isMutating(req)) {
      await db('system_logs').insert({
        level: 'info',
        traceId: req.traceId || null,
        userId: req.user?.id || null,
        subAccountId,
        action: `READ ${req.originalUrl}`,
        meta: JSON.stringify({ msg: 'Read access to protected account' }),
        created_at: new Date()
      });
    }

    next();
  } catch (err) {
    next(err);
  }
};

5) Retrieval endpoint (Express)

File: server/routes/laylaSuggest.js

const express = require('express');
const router = express.Router();
const { Pool } = require('pg');
const OpenAI = require('openai');
const pool = new Pool({ connectionString: process.env.DATABASE_URL });
const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

// feature-flag gated; demo only
router.get('/demo/layla-suggest', async (req,res,next) => {
  try {
    const { message_id } = req.query;
    // 1) fetch context
    const ctxReq = await pool.query('SELECT thread_id, body FROM messages WHERE id=$1', [message_id]);
    if (!ctxReq.rows[0]) return res.status(404).json({error:'message_not_found'});
    const thread_id = ctxReq.rows[0].thread_id;
    // 2) get inbound context
    const messages = await pool.query('SELECT id, body, direction FROM messages WHERE thread_id=$1 ORDER BY created_at ASC', [thread_id]);
    const conversation = messages.rows.map(r => `${r.direction.toUpperCase()}: ${r.body}`).join('\n');

    // 3) embed the context and run pgvector similarity
    const emb = await client.embeddings.create({ model: 'text-embedding-3-small', input: ctxReq.rows[0].body });
    const vec = emb.data[0].embedding;
    const k = 6;
    const pgRes = await pool.query(`SELECT message_id, meta FROM layla_embeddings ORDER BY embedding <#> $1 LIMIT $2`, [vec, k]);

    // 4) fetch the actual examples
    const ids = pgRes.rows.map(r => r.message_id);
    const examples = ids.length ? await pool.query('SELECT m1.body as inbound, m2.body as reply FROM messages m1 JOIN messages m2 ON m1.thread_id=m2.thread_id AND m2.direction=$2 WHERE m1.id = ANY($1::int[]) LIMIT 6', [ids, 'inbound']) : {rows:[]};

    const retrieved_examples = examples.rows.map(r => `IN: ${r.inbound}\nOUT: ${r.reply}`).join('\n\n');

    // 5) assemble prompt and call OpenAI
    const prompt = require('fs').readFileSync('./persona/layla_prompt_template.txt','utf8')
      .replace('{retrieved_examples}', retrieved_examples)
      .replace('{conversation_context}', conversation);

    const completion = await client.chat.completions.create({
      model: 'gpt-4o-mini', // choose availability
      messages: [{role:'system', content: prompt}],
      max_tokens: 120
    });
    const replyText = completion.choices[0].message.content.trim();
    // 6) return
    res.json({ reply: replyText, examples: retrieved_examples });
  } catch (err) { next(err); }
});

#!/usr/bin/env python3
# Fetch inbound->Layla_reply pairs, redact, write JSONL for training
import os, json, psycopg2, re
conn = psycopg2.connect(os.environ['DATABASE_URL'])
cur = conn.cursor()
cur.execute("""
  SELECT m1.id,m1.body as inbound, m2.body as reply
  FROM messages m1
  JOIN messages m2 ON m1.thread_id=m2.thread_id
  WHERE m1.sub_account_id=22 AND m1.direction='inbound' AND m2.direction='outbound' AND m2.sub_account_id=22
""")
rows = cur.fetchall()
def redact(s):
    s = re.sub(r'\b\\d{10,}\\b','[REDACTED]',s)
    s = re.sub(r'https?://\\S+','[LINK]',s)
    return s
out = []
for id,inbound,reply in rows:
  inbound_r = redact(inbound)
  reply_r = redact(reply)
  out.append({"prompt": inbound_r, "completion": reply_r})
with open('layla_train.jsonl','w') as f:
  for o in out: f.write(json.dumps(o)+'\\n')
print('Wrote', len(out))
2) LoRA training command (Hugging Face + PEFT)

File: train_lora.py (simplified — uses trl/transformers + peft)

# This is an example. In practice use the Hugging Face examples with accelerate + peft.
# See README below for exact commands.

from datasets import load_dataset
from peft import LoraConfig, get_peft_model
from transformers import AutoModelForCausalLM, AutoTokenizer, TrainingArguments, Trainer

model_name = "meta-llama/Llama-2-13b-chat-hf"  # choose appropriately
tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForCausalLM.from_pretrained(model_name, load_in_8bit=True, device_map='auto')

# load dataset
ds = load_dataset('json', data_files='layla_train.jsonl', split='train')

# tokenization, etc.
def preprocess(example):
    prompt = "### Instruction:\\n" + example['prompt'] + "\\n### Response:\\n"
    input_ids = tokenizer(prompt + example['completion'], truncation=True, max_length=512)
    return input_ids

ds = ds.map(preprocess)

# peft config
lora_config = LoraConfig(r=8, lora_alpha=32, target_modules=["q_proj","v_proj"], lora_dropout=0.05, bias="none")
model = get_peft_model(model, lora_config)

training_args = TrainingArguments(
    output_dir="lora-layla-v1",
    per_device_train_batch_size=4,
    num_train_epochs=3,
    learning_rate=2e-4,
    fp16=True,
    logging_steps=50
)

trainer = Trainer(model=model, args=training_args, train_dataset=ds)
trainer.train()
model.save_pretrained("lora-layla-v1")
