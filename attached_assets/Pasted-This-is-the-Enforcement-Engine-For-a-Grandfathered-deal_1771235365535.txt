### 🔱 The "Legacy Revocation" Webhook

This script listens to Stripe’s payment failure events. It gives them a 3-day "Grace Period," but if the final attempt fails, it wipes their Legacy status from the database.

```javascript
// api/webhooks/stripe-enforcement.ts
import { prisma } from '../lib/db';
import { sendUrgentAlert } from '../lib/notifications';

export async function handlePaymentFailure(event) {
  const subscription = event.data.object;
  const userId = subscription.metadata.userId;

  switch (event.type) {
    case 'invoice.payment_failed':
      // 1. THE WARNING: Tell them they are about to lose God Mode
      await sendUrgentAlert(userId, {
        title: "⚠️ LEGACY STATUS AT RISK",
        message: "Your payment failed. You have 72 hours to update your card or your 50% Grandfathered rate will be permanently revoked.",
        vibe: "emergency-red"
      });
      break;

    case 'customer.subscription.deleted':
      // 2. THE EXECUTION: The sub is canceled. Strip the status.
      const user = await prisma.user.findUnique({ where: { id: userId } });
      
      if (user.isGrandfathered) {
        await prisma.user.update({
          where: { id: userId },
          data: {
            isGrandfathered: false,
            legacyRateLocked: false,
            tier: "INACTIVE"
          }
        });

        // 3. THE LOG: Record the loss of status
        await prisma.auditLog.create({
          data: {
            userId,
            action: "LEGACY_STATUS_REVOKED",
            details: "Subscription lapsed. Grandfathered pricing removed."
          }
        });

        console.log(`❌ User ${userId} has lost Legacy status.`);
      }
      break;
  }
}

```
