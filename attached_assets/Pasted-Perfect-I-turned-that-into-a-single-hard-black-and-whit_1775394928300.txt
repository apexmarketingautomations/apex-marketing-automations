```js
// update_layla_telegram_config.js
const { Pool } = require('pg');
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

async function upsertTelegram() {
  const subAccountId = 22; // Layla
  const { rows } = await pool.query('SELECT operator_config FROM sub_accounts WHERE id=$1', [subAccountId]);
  if (!rows.length) throw new Error('sub_account not found ' + subAccountId);
  const existing = rows[0].operator_config || {};
  existing.telegram = {
    link: "https://t.me/LaylasLifeee",
    allowed: true,
    explicit_ask_phrases: [
      "where can i see more",
      "link to more",
      "send me your link",
      "how do i see more",
      "where do i find you",
      "where do i follow",
      "want more pics",
      "want more content",
      "show me more",
      "got more content",
      "send more",
      "how do i subscribe",
      "where do i get the extras"
    ],
    threshold_rules: {
      interest_score_needed: 3,
      score_rules: {
        explicit_interest_phrase: 1,
        positive_reply_to_probe: 1,
        emoji_enthusiasm_heart_fire: 1,
        repeated_interest_within_conv: 1
      }
    },
    templates: {
      telegram_send: "i drop extras there 💕 join me: t.me/LaylasLifeee — wanna come thru?",
      clarify_ambiguous: "ahh i might have more — what kinda stuff do u mean by 'more' so i send the right thing?"
    },
    safety: {
      block_on_payment_requests: true,
      block_on_explicit_for_pay: true,
      escalate_on_underage: true
    },
    audit: true
  };

  await pool.query('UPDATE sub_accounts SET operator_config = $1 WHERE id = $2', [existing, subAccountId]);
  console.log('Updated operator_config.telegram for Layla (sub_account_id=' + subAccountId + ')');
  await pool.end();
}

upsertTelegram().catch(e => { console.error(e); process.exit(1); });
```
