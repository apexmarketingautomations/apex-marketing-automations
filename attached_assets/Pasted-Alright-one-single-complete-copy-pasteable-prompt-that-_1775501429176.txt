````
```tsx
// client/src/components/MediaUpload.tsx
import React, { useState, useRef } from "react";

export default function MediaUpload({ subAccountId, onUploaded }:{
  subAccountId?: number;
  onUploaded?: (items:any)=>void;
}) {
  const [files,setFiles] = useState<File[]>([]);
  const [uploading,setUploading] = useState(false);
  const [error,setError] = useState<string|null>(null);
  const fileRef = useRef<HTMLInputElement|null>(null);

  function onSelect(e:React.ChangeEvent<HTMLInputElement>){
    const chosen = e.target.files ? Array.from(e.target.files) : [];
    setFiles(chosen);
  }

  async function upload(){
    if(!files.length) return;
    setUploading(true); setError(null);
    try{
      const fd = new FormData();
      files.forEach(f => fd.append("files", f));
      if(subAccountId) fd.append("sub_account_id", String(subAccountId));
      const res = await fetch("/api/media/upload",{ method:"POST", body:fd, credentials:"include" });
      if(!res.ok){
        const txt = await res.text();
        throw new Error(txt || `Upload failed: ${res.status}`);
      }
      const json = await res.json();
      setFiles([]);
      if(fileRef.current) fileRef.current.value = "";
      if(onUploaded) onUploaded(json);
    }catch(err:any){
      setError(err.message || "Upload failed");
    }finally{ setUploading(false); }
  }

  return (
    <div style={{border:"1px solid rgba(255,255,255,0.06)",padding:12,borderRadius:8,background:"rgba(255,255,255,0.02)"}}>
      <div style={{display:"flex",justifyContent:"space-between",alignItems:"center"}}>
        <div>
          <strong>Media library & upload</strong>
          <div style={{fontSize:12,color:"#9aa"}}>Upload images and videos for posts. Drag & drop or click to choose files.</div>
        </div>
        <div>
          <input ref={fileRef} type="file" multiple accept="image/*,video/*" onChange={onSelect} />
        </div>
      </div>

      {files.length>0 && (
        <div style={{marginTop:12,display:"grid",gridTemplateColumns:"repeat(auto-fit,minmax(120px,1fr))",gap:8}}>
          {files.map((f,i)=>(
            <div key={i} style={{border:"1px solid rgba(0,0,0,0.12)",padding:6,borderRadius:6,textAlign:"center"}}>
              <div style={{fontSize:12,fontWeight:600}}>{f.name}</div>
              <div style={{fontSize:11,color:"#999"}}>{Math.round(f.size/1024)} KB</div>
            </div>
          ))}
        </div>
      )}

      <div style={{marginTop:12,display:"flex",gap:8}}>
        <button onClick={upload} disabled={uploading || !files.length} className="btn btn-primary">
          {uploading ? "Uploading…" : "Upload"}
        </button>
        <button onClick={()=>{ setFiles([]); if(fileRef.current) fileRef.current.value=""; }} className="btn btn-ghost">Clear</button>
      </div>

      {error && <div style={{color:"#ff6b6b",marginTop:10}}>{error}</div>}
    </div>
  );
}
````

3. Create `server/routes/media.ts` with the exact code below (uses multer memory store, writes to local folder by default, checks `sub_accounts.is_protected`, logs to system_logs):

```ts
// server/routes/media.ts
import express from "express";
import multer from "multer";
import path from "path";
import fs from "fs/promises";
import { v4 as uuidv4 } from "uuid";

const router = express.Router();
const upload = multer({ storage: multer.memoryStorage(), limits: { fileSize: 50 * 1024 * 1024 } });
const UPLOAD_DIR = process.env.MEDIA_UPLOAD_DIR || "/tmp/uploads";
(async()=>{ try{ await fs.mkdir(UPLOAD_DIR,{recursive:true}); }catch(e){} })();

router.post("/upload", upload.array("files"), async (req, res) => {
  try {
    const db = (req as any).db ?? req.app.get("db");
    const subAccountId = req.body.sub_account_id ? Number(req.body.sub_account_id) : null;

    if(subAccountId){
      const r = await db.query("SELECT is_protected FROM sub_accounts WHERE id=$1", [subAccountId]);
      if(r.rows[0]?.is_protected){
        await db.query(`INSERT INTO system_logs (severity,module,message,metadata,created_at) VALUES ($1,$2,$3,$4,now())`, ['warn','media','upload_blocked_protected', JSON.stringify({subAccountId, user: req.user?.id||'unknown'})]);
        return res.status(403).json({ error: "protected_account" });
      }
    }

    const files = (req.files as Express.Multer.File[]) || [];
    const out:any[] = [];
    for(const f of files){
      const ext = path.extname(f.originalname) || "";
      const filename = `${Date.now()}-${uuidv4()}${ext}`;
      const target = path.join(UPLOAD_DIR, filename);
      await fs.writeFile(target, f.buffer);
      out.push({ originalName: f.originalname, filename, path: target, size: f.size, mime: f.mimetype });
      // optionally insert into media_items table here
    }

    await db.query(`INSERT INTO system_logs (severity,module,message,metadata,created_at) VALUES ($1,$2,$3,$4,now())`, ['info','media','media_uploaded', JSON.stringify({ count: out.length, subAccountId, uploadedBy: req.user?.id||null })]);

    res.json({ ok: true, uploaded: out });
  } catch (err) {
    console.error("media upload error", err);
    res.status(500).json({ error: String(err) });
  }
});

export default router;
```

```ts
// server/services/validation/isRepetitiveOpener.ts
export function isRepetitiveOpener(text?: string) {
  if (!text || typeof text !== "string") return false;
  const s = text.trim();
  if (s.length === 0) return false;
  const patterns = [
    /^\s*(you are (so )?beautiful\b)/i,
    /^\s*(i love you\b)/i,
    /^\s*(can i get your number\b)/i,
    /^\s*(you the baddest\b)/i,
    /^\s*(hey|hi|hello|babe|baby)[\s!,.]?$/i
  ];
  const words = s.split(/\s+/);
  if (words.length <= 3) {
    const first = words[0]?.toLowerCase();
    if (words.every(w => w.toLowerCase() === first)) return true;
  }
  if (/^[\p{Emoji_Presentation}\p{So}\p{Sk}]+$/u.test(s)) return true;
  return patterns.some(re => re.test(s));
}
```
