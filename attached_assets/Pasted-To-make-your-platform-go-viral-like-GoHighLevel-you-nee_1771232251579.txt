### 1. The Affiliate Database Schema

You need to track who referred whom and how much they are owed.

```prisma
// Add these to your schema.prisma

model AffiliateProfile {
  id             String   @id @default(cuid())
  userId         String   @unique
  user           User     @relation(fields: [userId], references: [id])
  affiliateCode  String   @unique // e.g., "AGM_LAUNCH"
  commissionRate Float    @default(0.40) // 40%
  totalEarned    Float    @default(0.0)
  pendingPayout  Float    @default(0.0)
  referrals      User[]   @relation("Referrals")
}

// Update User model to include a referrer
model User {
  // ... existing fields
  referrerId     String?
  referrer       AffiliateProfile? @relation("Referrals", fields: [referrerId], references: [id])
}

```

### 3. The "Payout" Engine (Fintech Logic)

This script runs whenever a customer pays their monthly bill. It automatically splits the money and moves the commission to the Affiliate's "Pending" bucket.

```javascript
// lib/affiliate-payouts.ts

export async function processCommission(userId, paymentAmount) {
  // 1. Find the user who just paid
  const user = await prisma.user.findUnique({
    where: { id: userId },
    include: { referrer: true }
  });

  // 2. If they have a referrer, calculate the cut
  if (user.referrer) {
    const commission = paymentAmount * user.referrer.commissionRate;

    // 3. Update the Affiliate's balance
    await prisma.affiliateProfile.update({
      where: { id: user.referrer.id },
      data: {
        totalEarned: { increment: commission },
        pendingPayout: { increment: commission }
      }
    });

    // 4. Log the transaction for the Affiliate Dashboard
    await prisma.transaction.create({
      data: {
        affiliateId: user.referrer.id,
        amount: commission,
        note: `Commission from ${user.businessName} payment`
      }
    });
  }
}

```
